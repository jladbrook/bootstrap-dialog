Examples for studious monkeys

Construct a dialog

var dialog = new BootstrapDialog({
	title	:	'What do you think of Orange?',
	content	:	'So, tell me please?',
	buttons	:	[{
		label	:	'I\'ve no idea.',
		onclick	:	function(dialog){
			dialog.close();
		}
	}, {
		label	:	'He\'s a jerk!',
		cssClass:	'btn-primary',
		onclick	:	function(dialog){
			alert('OK I will tell him.');
			dialog.close();
		}
	}, {
		label	:	'I like him.',
		cssClass:	'btn-danger',
		onclick	:	function(dialog){
			alert('Ohhhh.. I can\'t believe that.');
			dialog.close();
		}
	}]
});
dialog.open();

Ultimate example!

The example below shows how to create fancy contents in the dialog and manipulate the title, content, and buttons.

Question Dialog

var dialog = new BootstrapDialog();
// Both of title and content accept plain text or jquery object, or if a function was provided a dialog instance will be passed to the function.  
dialog.setTitle(function(dialog){
	var $title = $('<div></div>');
	$title.append('<span>Please anwser the questions.</span>');
	var $leaveNow = $('<a href="#">Leave now</a>');
	$leaveNow.click({dialog: dialog}, function(event){
		// You can click the link in the title to close the dialog.
		event.data.dialog.close();
		event.preventDefault();
	});
	$title.append($leaveNow);
	
	return $title;
});
dialog.setContent(function(dialog){
	var $content = $('<div></div>');
	$content.append('<p class="help-block">I understand that the question may be difficult for you but please try to answer it anyway.</p>');
	$content.append('<hr>');
	$content.append('<label>Question: 1 + 1 = ?</label>');
	$input = $('<input type="text"/>');
	$content.append($input);
	$content.append('<div class="hidden-block hide"><img src="http://www.ilex-press.com/wp-content/uploads/pig-thumb1.jpg" width="200" /><br>I can even answer this.</div>');
	$input.keyup({dialog: dialog}, function(event){
		var dialog = event.data.dialog;
		var value = $(this).val();
		if(Number(value) == 2){
			// Enable the button and let the piggy hide when the user provide a correct answer
			dialog.getButton('btn-close').removeAttr('disabled');
			dialog.getBody().find('.hidden-block').hide();
			// Store data in dialog instance, then you can use if after the dialog is popped down.
			dialog.setData('last-answer', value);
		}else{
			dialog.getButton('btn-close').attr('disabled', 'disabled');
			dialog.getBody().find('.hidden-block').show();
		}
	});
	
	return $content;
});
dialog.setButtons([{
	id		:		'btn-close',
	label	:		'Enable this button only when you answered the question correctly.',
	cssClass:		'btn-primary',
	onclick	:		function(dialog){
		// Demonstrate getting information of the dialog
		alert('Good, you answered the question correctly!');
		alert('The html of dialog header is: ' + dialog.getHeader().html());
		alert('The html of dialog body is: ' + dialog.getBody().html());
		alert('The html of dialog footer is: ' + dialog.getFooter().html());
		alert('The value you stored in dialog is: ' + dialog.getData('last-answer'));
		dialog.close();
	}
}]);
dialog.setReady(function(dialog){
	// Disable the button before the dialog is popped up
	dialog.getButton('btn-close').attr('disabled', 'disabled');
});
dialog.open();
